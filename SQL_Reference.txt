STUDENT INFORMATION SYSTEM - SQL QUERY REFERENCE
=================================================


--- STORED PROCEDURE ---

-- Inserts a new student into Student then routes them to Student_FullTime or Student_PartTime.
CALL EnrolStudent(p_FirstName, p_LastName, p_PhoneNumber, p_DateOfBirth, p_Email,
  p_GradeLevel, p_EnrolmentStatus, p_StudentType, p_ExtraCurricular,
  p_GuardianContactInfo, p_ReasonForPartTime, p_HoursEnrolledPerWeek, p_PasswordHash);


--- DDL: CREATE TABLES ---

-- Stores school term periods (Fall, Spring, Summer, Winter) with start and end dates.
CREATE TABLE academic_term (term_id, term_name, school_year, start_date, end_date);

-- Stores available courses with subject area, grade level, and description.
CREATE TABLE course (course_id, course_name, subject_area, grade_level, course_description);

-- Links a course to a faculty member, schedule slot, and academic term.
CREATE TABLE course_offering (course_offering_id, section_name, course_id, term_id, time_slot_id, faculty_id);

-- Tracks which students are enrolled in which course offerings and their status.
CREATE TABLE enrollment (enrollment_id, student_id, course_offering_id, enrollment_date, enrollment_status);

-- Stores faculty/staff records including bcrypt-hashed passwords and supervisor references.
CREATE TABLE faculty (faculty_id, first_name, last_name, email, password_hash, phone_number, position, supervisor_id);

-- Defines which courses require another course as a prerequisite.
CREATE TABLE prerequisite (course_id, prereq_course_id);

-- Stores weekly time slot patterns (e.g. MWF, TTh) with start and end times.
CREATE TABLE schedule (time_slot_id, days_of_week, start_time, end_time);

-- Stores student records including grade level, enrolment status, and bcrypt-hashed password.
CREATE TABLE student (StudentID, FirstName, LastName, PhoneNumber, DateOfBirth, Email, password_hash, GradeLevel, EnrolmentStatus);

-- Subclass table for full-time students storing extracurricular activities and guardian contact.
CREATE TABLE student_fulltime (StudentID, ExtraCurricularActivities, GuardianContactInfo);

-- Subclass table for part-time students storing reason and hours per week.
CREATE TABLE student_parttime (StudentID, ReasonForPartTime, HoursEnrolledPerWeek);

-- Junction table recording which faculty member supervises another.
CREATE TABLE supervises (supervisor_id, supervisee_id);


--- SELECT ---

-- Fetches a faculty member by email for login authentication.
SELECT faculty_id, first_name, last_name, email, password_hash, position FROM Faculty WHERE email = ?;

-- Fetches a student by email for login authentication, including their grade level.
SELECT StudentID, FirstName, LastName, Email, password_hash, GradeLevel FROM Student WHERE Email = ?;

-- Fetches all students.
SELECT * FROM Student;

-- Fetches a single student by ID.
SELECT * FROM Student WHERE StudentID = ?;

-- Fetches all faculty members.
SELECT * FROM Faculty;

-- Fetches a single faculty member by ID.
SELECT * FROM Faculty WHERE faculty_id = ?;

-- Fetches all courses.
SELECT * FROM Course;

-- Fetches a single course by ID.
SELECT * FROM Course WHERE course_id = ?;

-- Fetches all academic terms.
SELECT * FROM Academic_Term;

-- Fetches a single academic term by ID.
SELECT * FROM Academic_Term WHERE term_id = ?;

-- Fetches all schedule time slots.
SELECT * FROM Schedule;

-- Fetches a single schedule time slot by ID.
SELECT * FROM Schedule WHERE time_slot_id = ?;

-- Fetches all course offerings with joined course, term, schedule, and faculty data including grade level.
SELECT co.course_offering_id, co.section_name, c.course_name, c.subject_area, c.grade_level,
  at2.term_name, at2.school_year, s.days_of_week, s.start_time, s.end_time,
  f.first_name AS faculty_first_name, f.last_name AS faculty_last_name
FROM Course_Offering co
JOIN Course c ON co.course_id = c.course_id
JOIN Academic_Term at2 ON co.term_id = at2.term_id
JOIN Schedule s ON co.time_slot_id = s.time_slot_id
JOIN Faculty f ON co.faculty_id = f.faculty_id;

-- Fetches a single course offering by ID (same JOINs as above).
... WHERE co.course_offering_id = ?;

-- Fetches all enrollments with student name, course, section, term, schedule, and faculty data.
SELECT e.enrollment_id, e.student_id, e.enrollment_date, e.enrollment_status,
  s.FirstName AS student_first_name, s.LastName AS student_last_name,
  co.course_offering_id, co.section_name, c.course_name, c.subject_area,
  at2.term_name, at2.school_year, sch.days_of_week, sch.start_time, sch.end_time,
  f.first_name AS faculty_first_name, f.last_name AS faculty_last_name
FROM Enrollment e
JOIN Student s ON e.student_id = s.StudentID
JOIN Course_Offering co ON e.course_offering_id = co.course_offering_id
JOIN Course c ON co.course_id = c.course_id
JOIN Academic_Term at2 ON co.term_id = at2.term_id
JOIN Schedule sch ON co.time_slot_id = sch.time_slot_id
JOIN Faculty f ON co.faculty_id = f.faculty_id;

-- Fetches a single enrollment record by ID (same JOINs as above).
... WHERE e.enrollment_id = ?;

-- Fetches all prerequisites with both course names resolved via double JOIN on Course.
SELECT p.course_id, c1.course_name, p.prereq_course_id, c2.course_name AS prereq_course_name
FROM Prerequisite p
JOIN Course c1 ON p.course_id = c1.course_id
JOIN Course c2 ON p.prereq_course_id = c2.course_id;

-- Fetches all supervision relationships with both supervisor and supervisee names.
SELECT sv.supervisor_id, f1.first_name AS supervisor_first_name, f1.last_name AS supervisor_last_name,
  sv.supervisee_id, f2.first_name AS supervisee_first_name, f2.last_name AS supervisee_last_name
FROM Supervises sv
JOIN Faculty f1 ON sv.supervisor_id = f1.faculty_id
JOIN Faculty f2 ON sv.supervisee_id = f2.faculty_id;
